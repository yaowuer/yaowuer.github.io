#+TITLE: Javascript


* Prototype-Based OO

原型链:

#+BEGIN_SRC js
  var a = 222;
  var b = "333";
  var c = true;

  var tom = {
      addr: "珠海",
      name: "TOM",
      age: 18,
      weixin: "110",
      intro: function () { alert("hello, i am " + this.name); }
  };

  var jerry = {
      addr: "珠海",
      name: "Jerry",
      age: 14,
      weixin: "1ksfjk",
      intro: function () { alert("hello"); }
  };

  var class152 = {
      banzhuren: jerry,
      tiyulaoshi: tom
  };

  // Prototype-Based Object

  var badtom = {
      type: "bad",
      __proto__: tom
  }
  var goodtom = {
      type: "good",
      __proto__: tom
  }
  var sonofbadtom = {
      name: "tomson",
      age: 1,
      __proto__: badtom
  }
#+END_SRC

JS 中没有类，class 关键词只是一种语法糖（Syntax Sugar）。
